// Example of memory management in C++
int main() {
    // Stack allocation - automatically managed
    int stackVar = 5;
    
    // Heap allocation - manually managed
    int* heapVar = new int(10);
    
    // Using the heap-allocated variable
    std::cout << *heapVar << std::endl;  // Output: 10
    
    // Must deallocate to prevent memory leaks
    delete heapVar;
    
    return 0;
}

Arrays: The Foundation of Data Structures

Arrays are the most fundamental data structure and form the basis for many others. Understanding array manipulation is crucial for solving algorithmic problems.

Key Concepts

  • Static arrays vs Dynamic arrays (vectors)
  • Two-pointer technique
  • Sliding window pattern
  • Prefix sums and range queries
// Two-pointer technique example
bool hasPairWithSum(std::vector& arr, int targetSum) {
    std::sort(arr.begin(), arr.end());  // Sort the array first
    
    int left = 0;
    int right = arr.size() - 1;
    
    while (left < right) {
        int currentSum = arr[left] + arr[right];
        
        if (currentSum == targetSum) {
            return true;
        } else if (currentSum < targetSum) {
            left++;  // Need a larger value
        } else {
            right--;  // Need a smaller value
        }
    }
    
    return false;
}
