// Two-pointer technique example
bool hasPairWithSum(std::vector& arr, int targetSum) {
    std::sort(arr.begin(), arr.end());  // Sort the array first
    
    int left = 0;
    int right = arr.size() - 1;
    
    while (left < right) {
        int currentSum = arr[left] + arr[right];
        
        if (currentSum == targetSum) {
            return true;
        } else if (currentSum < targetSum) {
            left++;  // Need a larger value
        } else {
            right--;  // Need a smaller value
        }
    }
    
    return false;
}

Linked Lists: Dynamic Data Structures

Linked lists are linear data structures where elements are stored in nodes, and each node points to the next node in the sequence. Unlike arrays, linked lists don't require contiguous memory allocation, making insertions and deletions more efficient.

Linked List Structure

Key Concepts

  • Node Structure: Each node contains data and a reference to the next node
  • Singly vs Doubly: Singly linked lists have only next pointers, while doubly linked lists have both next and previous pointers
  • Operations: Insertion, deletion, traversal, searching
  • Fast & Slow Pointers: A technique to solve many linked list problems
// Singly Linked List Node in C++
struct Node {
    int data;        // Data stored in the node
    Node* next;      // Pointer to the next node
    
    // Constructor for easy node creation
    Node(int val) : data(val), next(nullptr) {}
};

// Creating a simple linked list with 3 nodes
Node* createLinkedList() {
    // Create nodes
    Node* head = new Node(10);
    Node* second = new Node(20);
    Node* third = new Node(30);
    
    // Connect nodes
    head->next = second;
    second->next = third;
    // third->next is already nullptr from constructor
    
    return head;
}

// Function to traverse and print a linked list
void printList(Node* head) {
    Node* current = head;
    while (current != nullptr) {
        cout << current->data << " -> ";
        current = current->next;
    }
    cout << "nullptr" << endl;
}

Advantages and Disadvantages

Advantages
  • Dynamic size (grows/shrinks at runtime)
  • Efficient insertions and deletions (O(1) if position is known)
  • No need for contiguous memory allocation
  • Can be easily implemented for stacks and queues
Disadvantages
  • No random access (O(n) to access arbitrary elements)
  • Extra memory for storing pointers
  • Not cache-friendly due to non-contiguous memory
  • Requires pointers, making it more complex than arrays

Basic Operations
class SinglyLinkedList {
private:
    Node* head;
    
public:
    // Constructor
    SinglyLinkedList() : head(nullptr) {}
    
    // Insert at beginning - O(1)
    void insertAtBeginning(int val) {
        Node* newNode = new Node(val);
        newNode->next = head;
        head = newNode;
    }
    
    // Insert at end - O(n)
    void insertAtEnd(int val) {
        Node* newNode = new Node(val);
        
        // If list is empty
        if (head == nullptr) {
            head = newNode;
            return;
        }
        
        // Find the last node
        Node* current = head;
        while (current->next != nullptr) {
            current = current->next;
        }
        
        // Link the last node to new node
        current->next = newNode;
    }
    
    // Delete a node with given value - O(n)
    void deleteNode(int key) {
        // If list is empty
        if (head == nullptr) return;
        
        // If head has the key
        if (head->data == key) {
            Node* temp = head;
            head = head->next;
            delete temp;
            return;
        }
        
        // Find the node before the one to be deleted
        Node* current = head;
        while (current->next != nullptr && current->next->data != key) {
            current = current->next;
        }
        
        // If key not found
        if (current->next == nullptr) {
            return;
        }
        
        // Delete the node
        Node* temp = current->next;
        current->next = current->next->next;
        delete temp;
    }
}

Easy

Reverse a singly linked list.

Approach:

Use three pointers (prev, current, next) to reverse links one by one.

Node* reverseList(Node* head) {
    Node* prev = nullptr;
    Node* current = head;
    Node* next = nullptr;
    
    while (current != nullptr) {
        // Store next
        next = current->next;
        
        // Reverse current node's pointer
        current->next = prev;
        
        // Move pointers one position ahead
        prev = current;
        current = next;
    }
    
    // New head is the previous last node
    return prev;
}

Dynamic Programming: Breaking Down Complex Problems

Dynamic Programming (DP) is a powerful technique for solving problems by breaking them down into overlapping subproblems. By storing solutions to subproblems, DP avoids redundant calculations and improves efficiency.

Two Key Principles of Dynamic Programming
1. Optimal Substructure

The optimal solution to a problem can be constructed from optimal solutions of its subproblems.

2. Overlapping Subproblems

The problem can be broken down into subproblems that are reused multiple times.

Approaches to Dynamic Programming

Top-Down Approach (Memoization)

Start with the original problem and recursively break it down into subproblems, storing results to avoid recalculation.

// Fibonacci using memoization (top-down DP)
int fibMemo(int n, vector& memo) {
    // Base cases
    if (n <= 1) return n;
    
    // Check if already computed
    if (memo[n] != -1) return memo[n];
    
    // Calculate and store result
    memo[n] = fibMemo(n - 1, memo) + fibMemo(n - 2, memo);
    return memo[n];
}

int fibonacci(int n) {
    vector memo(n + 1, -1);
    return fibMemo(n, memo);
}
Bottom-Up Approach (Tabulation)

Start with the smallest subproblems and build up to the original problem, filling a table of results.

// Fibonacci using tabulation (bottom-up DP)
int fibTabulation(int n) {
    // Base cases
    if (n <= 1) return n;
    
    // Create and initialize DP table
    vector dp(n + 1);
    dp[0] = 0;
    dp[1] = 1;
    
    // Fill table bottom-up
    for (int i = 2; i <= n; i++) {
        dp[i] = dp[i - 1] + dp[i - 2];
    }
    
    return dp[n];
}

Common DP Problem Types

1. Optimization Problems
  • Knapsack Problem
  • Longest Common Subsequence
  • Longest Increasing Subsequence
  • Coin Change (minimum coins)
2. Counting Problems
  • Ways to reach a destination
  • Ways to count change
  • Number of distinct subsequences
  • Number of ways to decode a message
3. Path Problems
  • Minimum/Maximum Path Sum
  • Unique Paths in a Grid
  • Edit Distance
  • Shortest/Longest Common Supersequence
Visual Example: Climbing Stairs Problem

How many distinct ways can you climb n stairs if you can take 1 or 2 steps at a time?

// Dynamic Programming solution for Climbing Stairs
int climbStairs(int n) {
    if (n <= 2) return n;
    
    // Create DP table
    vector dp(n + 1);
    
    // Base cases
    dp[0] = 1;  // One way to climb 0 stairs (stay where you are)
    dp[1] = 1;  // One way to climb 1 stair (take one step)
    
    // Fill the table bottom-up
    for (int i = 2; i <= n; i++) {
        dp[i] = dp[i - 1] + dp[i - 2];
    }
    
    return dp[n];
}

DP Fundamentals: Memoization vs Tabulation

75 min

Learn the two main approaches to dynamic programming and when to use each.

Approach Comparison
Memoization (Top-Down)
  • Starts with the original problem
  • Uses recursion with a cache
  • Solves only necessary subproblems
  • Easier to implement for complex state transitions
  • May cause stack overflow for large inputs
Tabulation (Bottom-Up)
  • Starts with the smallest subproblems
  • Uses iteration to fill a table
  • Solves all subproblems in order
  • Often more efficient (no recursive overhead)
  • Sometimes harder to implement
Example: Fibonacci with Both Approaches
// Top-down approach with memoization

    // Compute and store result

    return memo[n];
}

int fib(int n) {
    if (n <= 1) return n;
    vector memo(n + 1, -1);
    return fibonacci(n, memo);
}

    dp[0] = 0;
    dp[1] = 1;

int lengthOfLIS(vector& nums) {
    int n = nums.size();
    if (n == 0) return 0;
    
    // dp[i] = length of LIS ending at index i
    vector dp(n, 1);
    
    // Fill dp table
    for (int i = 1; i < n; i++) {
        for (int j = 0; j < i; j++) {
            if (nums[i] > nums[j]) {
                dp[i] = max(dp[i], dp[j] + 1);
            }
        }
    }
    
    // Find maximum value in dp table
    return *max_element(dp.begin(), dp.end());
}

int longestCommonSubsequence(string text1, string text2) {
    int m = text1.length();
    int n = text2.length();
    
    // Create DP table
    vector> dp(m + 1, vector(n + 1, 0));
    
    // Fill the DP table
    for (int i = 1; i <= m; i++) {
        for (int j = 1; j <= n; j++) {
            if (text1[i - 1] == text2[j - 1]) {
                dp[i][j] = dp[i - 1][j - 1] + 1;
            } else {
                dp[i][j] = max(dp[i - 1][j], dp[i][j - 1]);
            }
        }
    }
    
    return dp[m][n];
}

322. Coin Change
Medium

You are given an integer array coins representing coins of different denominations and an integer amount representing a total amount of money. Return the fewest number of coins that you need to make up that amount.

Approach:

Use a 1D DP array where dp[i] represents the minimum number of coins needed to make amount i.

int coinChange(vector& coins, int amount) {
    // Initialize with amount+1 (representing "infinity")
    vector dp(amount + 1, amount + 1);
    dp[0] = 0;  // Base case: 0 coins needed to make 0 amount
    
    for (int i = 1; i <= amount; i++) {
        for (int coin : coins) {
            if (coin <= i) {
                dp[i] = min(dp[i], dp[i - coin] + 1);
            }
        }
    }
    
    return dp[amount] > amount ? -1 : dp[amount];
}
